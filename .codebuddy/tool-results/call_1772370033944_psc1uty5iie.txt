190: /**
191:  * Initialize all OpenClaw modules with default configuration
192:  */
193: export function initializeOpenClawModules(config: {
194:   builtInTools?: string[];
195:   policyHierarchy?: PolicyHierarchy;
196:   compactionConfig?: CompactionConfig;
197:   authProfiles?: AuthProfile[];
198:   memoryConfig?: MemorySearchConfig;
199:   allowlist?: AllowlistConfig;
200: } = {}): void {
201:   // Initialize Tool Policy
202:   if (config.policyHierarchy) {
203:     _getToolPolicyEngine(config.policyHierarchy);
204:   }
205: 
206:   // Initialize Compaction Engine
207:   if (config.compactionConfig) {
208:     _getSmartCompactionEngine(config.compactionConfig);
209:   }
210: 
211:   // Initialize Retry Engine
212:   if (config.authProfiles) {
213:     _getRetryFallbackEngine(config.authProfiles);
214:   }
215: 
216:   // Initialize Memory Search
217:   if (config.memoryConfig) {
218:     _getSemanticMemorySearch(config.memoryConfig);
219:   }
220: 
221:   // Initialize Plugin Conflict Detector
222:   _getPluginConflictDetector(config.builtInTools, config.allowlist);
223: }

228: export function resetAllOpenClawModules(): void {

231:   _resetSmartCompactionEngine();
232:   _resetRetryFallbackEngine();
233:   _resetSemanticMemorySearch();
234:   _resetPluginConflictDetector();
235: }
