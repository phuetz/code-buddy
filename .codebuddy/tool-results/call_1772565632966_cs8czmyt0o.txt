176:   /**
177:    * Tool Result Compaction Guard (OpenClaw / Manus AI #13)
178:    *
179:    * Before each model call, scan accumulated tool result messages.
180:    * If their total size exceeds the threshold (default 70K chars ≈ ~17K tokens),
181:    * compress the oldest ones using RestorableCompressor — replacing full content
182:    * with a compact stub referencing the callId. The content remains restorable
183:    * via the `restore_context` tool.
184:    *
185:    * This prevents deep agent chains from silently overflowing the context window.
186:    */
187:   private compactLargeToolResults(
188:     preparedMessages: CodeBuddyMessage[],
189:     maxToolResultChars = 70_000
190:   ): CodeBuddyMessage[] {
191:     // Sum characters from tool result messages
192:     let totalToolChars = 0;
193:     for (const m of preparedMessages) {
194:       if (m.role === 'tool' && typeof m.content === 'string') {
195:         totalToolChars += m.content.length;
196:       }
197:     }
198: 
199:     if (totalToolChars <= maxToolResultChars) return preparedMessages;
200: 
201:     const compressor = getRestorableCompressor();
202:     // Compress oldest tool results first (front of the list)
203:     const result = [...preparedMessages];
204:     let charsToFree = totalToolChars - maxToolResultChars;
205: 
206:     for (let i = 0; i < result.length && charsToFree > 0; i++) {
207:       const m = result[i];
208:       if (m.role === 'tool' && typeof m.content === 'string' && m.content.length > 500) {
209:         const callId = (m as { tool_call_id?: string }).tool_call_id || `tool_${i}`;
210:         const compressed = compressor.compress([{
211:           role: m.role,
212:           content: m.content,
213:           tool_call_id: callId,
214:         }]);
215:         if (compressed.messages[0]) {
216:           charsToFree -= (m.content.length - (compressed.messages[0].content?.length ?? 0));
217:           result[i] = { ...m, content: compressed.messages[0].content ?? m.content };
218:         }
219:       }
220:     }
221: 
222:     logger.debug(`ToolResultCompactionGuard: compacted tool results`, {
223:       before: totalToolChars,
224:       freed: totalToolChars - charsToFree,
225:     });
226: 
227:     return result;
228:   }
