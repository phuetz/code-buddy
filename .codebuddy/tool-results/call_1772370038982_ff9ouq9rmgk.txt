62: async function lazyLoad<T>(key: string, loader: () => Promise<T>): Promise<T> {
63:   if (lazyModuleCache.has(key)) {
64:     return lazyModuleCache.get(key) as T;
65:   }
66:   const startTime = PERF_TIMING ? Date.now() : 0;
67:   const module = await loader();
68:   if (PERF_TIMING) {

70:     if (loadTime > 50) { // Only log slow loads
71:       recordStartupPhase(`lazy:${key} (${loadTime}ms)`);
72:     }
73:   }
74:   lazyModuleCache.set(key, module);
75:   return module;
76: }
